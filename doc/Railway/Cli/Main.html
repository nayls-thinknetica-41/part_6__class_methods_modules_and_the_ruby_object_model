def control(commands)
  case commands
  when 'i', 'inspect'
    pp Cli.state
  when 'q', 'quit'
    exit 0
  when 's', 'station'
    Railway::Cli::Station.help_and_control
  when 't', 'train'
    Railway::Cli::Train.help_and_control
  when 'r', 'route'
    Railway::Cli::Route.help_and_control
  when 'w', 'wagon'
    Railway::Cli::Wagon.help_and_control
  else
    help
  end
end
